# flying feather 
name feather
type none
gfxshadow 1
shadow 0

animationscript data/scripts/vars/constants.h

globalmap 1

#type 0
anim spawn
loop 1
offset 23 22
delay 10
@script
void self = getlocalvar("self");
if(rand()%2==0){changeentityproperty(self, "animation", A_IDLE);}
else{changeentityproperty(self, "animation", A_SLEEP);}
if(getentityproperty(self, "a")<0)
{killentity(self);}
@end_script
frame data/bgs/fiend/feather/1.gif

anim idle

offset 23 22
delay 10
@script
void self = getlocalvar("self");

{killentity(self);}
@end_script
